━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. ИНИЦИАЛИЗАЦИЯ И НАСТРОЙКА
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  git init
  ────────
  Создаёт новый пустой репозиторий Git в текущей папке.
  Появится скрытая папка .git — там будет вся история проекта.
  Флаги:
    --bare    — создать «голый» репозиторий (без рабочей копии,
                используется на серверах)
    -q, --quiet — не выводить сообщения

  git clone <url> [папка]
  ──────────────────────
  Скачивает готовый репозиторий с удалённого сервера (GitHub,
  GitLab и т.д.) вместе со всей историей.
  Флаги:
    --depth <n>   — склонировать только последние n коммитов
                    (глубокое клонирование, экономит трафик)
    --branch <b>  — сразу переключиться на ветку <b>
    --single-branch — клонировать только одну ветку
    -b <name>     — создать ветку с указанным именем после клонирования

  git config
  ──────────
  Настройка параметров Git (имя, почта, редактор и т.д.).
  Флаги:
    --global   — применить настройку для всего пользователя
                (хранится в ~/.gitconfig)
    --local    — только для текущего репозитория (по умолчанию)
    --system   — для всех пользователей системы
    --list     — показать все текущие настройки
    --edit     — открыть конфиг в текстовом редакторе

  Примеры:
    git config --global user.name "Иван Иванов"
    git config --global user.email "ivan@example.com"

2. ОСНОВНАЯ РАБОТА С ФАЙЛАМИ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  git status
  ──────────
  Показывает текущее состояние репозитория: какие файлы
  изменены, какие добавлены в «индекс» (staging area),
  какие не отслеживаются.
  Флаги:
    -s, --short — сокращённый вывод (одна строка на файл)
    -b, --branch — показать имя текущей ветки
    --ignored   — показать игнорируемые файлы

  git add <файл/папка>
  ─────────────────────
  Добавляет изменения в индекс (staging area) — подготавливает
  их к коммиту. Без этого Git не запомнит изменения.
  Флаги:
    -A, --all   — добавить все изменения (новые, изменённые,
                  удалённые файлы)
    .           — добавить всё в текущей папке
    -p, --patch — добавлять только отдельные фрагменты изменений
                  (удобно для частичного коммита)
    -n, --dry-run — показать, что БУДЕТ добавлено, но не добавлять
    -f, --force — добавить файлы, проигнорированные в .gitignore

  git commit
  ──────────
  Создаёт коммит (сохранение / снимок) из того, что сейчас
  находится в индексе (staging area).
  Флаги:
    -m "<текст>" — указать сообщение коммита прямо в команде
                   (без неё откроется редактор)
    -a           — автоматически добавить все отслеживаемые
                   изменённые файлы (git add + git commit)
    --amend      — исправить последний коммит (если забыли файл
                   или хотите поменять сообщение)
    --no-edit    — не менять сообщение при --amend
    -v           — показать diff в сообщении коммита (подробно)

  git rm <файл>
  ─────────────
  Удаляет файл из репозитория и из рабочей директории.
  Флаги:
    --cached    — удалить только из отслеживания Git,
                  файл в папке останется
    -r          — удалять папки рекурсивно
    -f, --force — принудительное удаление (если есть
                  незакоммиченные изменения)

  git mv <источник> <назначение>
  ──────────────────────────────
  Переименовывает или перемещает файл и сразу добавляет
  это изменение в индекс.

3. ПРОСМОТР ИСТОРИИ И ИЗМЕНЕНИЙ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  git log
  ───────
  Показывает историю коммитов в обратном хронологическом порядке
  (сначала свежие).
  Флаги:
    --oneline    — сокращённый вывод (один коммит = одна строка)
    --graph      — показать ASCII-граф ветвлений
    --all        — показать все ветки, а не только текущую
    -n <число>   — показать только последние N коммитов
    --author=<имя> — фильтр по автору
    --since="2024-01-01" — коммиты начиная с даты
    --until="2024-12-31" — коммиты до даты
    -p           — показать ещё и содержимое изменений (патч)
    --stat       — показать статистику изменённых файлов
    --grep="<текст>" — поиск по сообщению коммита
    -S "<текст>" — поиск коммитов, где появился/исчез указанный
                   текст (очень мощный инструмент)
    --follow <файл> — показать историю файла, включая переименования

  git diff
  ────────
  Показывает разницу между версиями файлов.
  Флаги:
    без флагов — разница между рабочей директорией и индексом
    --staged / --cached — разница между индексом и последним
                          коммитом (что пойдёт в коммит)
    HEAD        — разница между рабочей директорией и последним
                  коммитом
    <коммит1> <коммит2> — разница между двумя коммитами
    -- <файл>   — показать diff только для одного файла
    --stat      — только статистика (сколько строк добавлено/удалено)
    --word-diff — показывать изменения по словам, а не по строкам
    --color-words — цветная подсветка изменений по словам
    -w, --ignore-all-space — игнорировать пробелы

  git show <коммит>
  ──────────────────
  Показывает информацию о конкретном коммите: автор, дата,
  сообщение и сами изменения.
  Флаги:
    --stat      — только статистика
    --name-only — только имена изменённых файлов
    -m          — при merge-коммитах показывать diff по каждому
                  родителю отдельно

  git blame <файл>
  ─────────────────
  Показывает построчно, кто и в каком коммите последний раз
  менял каждую строку файла. Помогает понять, кто написал
  конкретную строчку кода и зачем.
  Флаги:
    -L <start>,<end> — показать только указанные строки
    -w          — игнорировать изменения пробелов

  git shortlog
  ────────────
  Группирует коммиты по автору — удобно для генерации changelog.
  Флаги:
    -s, --summarize — только количество коммитов на автора
    -n          — сортировать по количеству коммитов
    --since, --until — ограничение по дате

4. ВЕТКИ (BRANCHES)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  git branch
  ──────────
  Управление ветками. Без флагов — показывает список веток
  (звёздочкой отмечена текущая).
  Флаги:
    <имя>         — создать новую ветку
    -d <имя>      — удалить ветку (только если она слита)
    -D <имя>      — принудительно удалить ветку (слита или нет)
    -m <старое> <новое> — переименовать ветку
    -M <старое> <новое> — принудительно переименовать (если новое
                          имя уже занято)
    -a            — показать все ветки, включая удалённые
    -r            — показать только удалённые ветки
    --merged      — показать только ветки, которые уже слиты
    --no-merged   — показать ветки, которые ещё не слиты
    --contains <коммит> — ветки, содержащие указанный коммит

  git switch
  ──────────
  Переключиться на другую ветку (современная команда, появилась
  в Git 2.23 вместо git checkout для веток).
  Флаги:
    -c <имя>    — создать новую ветку и сразу переключиться на неё
    -C <имя>    — принудительно создать/пересоздать ветку
    -           — вернуться на предыдущую ветку
    --detach    — переключиться в «отделённый» HEAD (без ветки)

  git checkout
  ────────────
  Старая, но всё ещё полезная команда. Может переключать ветки
  и восстанавливать файлы.
  Флаги:
    -b <имя>    — создать ветку и переключиться на неё
    -B <имя>    — создать/пересоздать ветку принудительно
    -- <файл>   — откатить файл до состояния в индексе
    HEAD -- <файл> — откатить файл до последнего коммита
    --ours      — при конфликте оставить нашу версию файла
    --theirs    — при конфликте взять их версию файла

  git merge <ветка>
  ──────────────────
  Сливает указанную ветку в текущую. Создаёт merge-коммит,
  если нет конфликтов.
  Флаги:
    --no-ff     — создать merge-коммит, даже если слияние
                  может быть fast-forward
    --ff-only   — слить только если возможно fast-forward
                  (иначе ошибка)
    --squash    — «сплющить» все коммиты сливаемой ветки
                  в один коммит без создания merge-коммита
    --abort     — отменить слияние при конфликтах
    --continue  — продолжить слияние после разрешения конфликтов
    --no-commit — сделать слияние, но не создавать коммит
                  (позволит доработать)
    --strategy=recursive — стратегия слияния (recursive —
                           стандарт для двух веток)
    --strategy=ours — автоматически брать нашу версию при
                      конфликтах

  git rebase
  ──────────
  Перемещает коммиты текущей ветки на верхушку другой ветки.
  В отличие от merge — не создаёт merge-коммит, а переписывает
  историю, делая её линейной.
  ВАЖНО: не использовать на публичных/общих ветках!
  Флаги:
    -i            — интерактивный rebase (можно переставлять,
                    объединять, удалять и править коммиты)
    --onto <база> — переместить коммиты на другую базу
    --continue    — продолжить после разрешения конфликта
    --skip        — пропустить проблемный коммит
    --abort       — отменить rebase целиком
    --autosquash  — автоматически объединять fixup/squash-коммиты
                    (нужно правильно называть коммиты)

  git cherry-pick <коммит>
  ─────────────────────────
  Берёт один конкретный коммит из другой ветки и применяет
  его изменения в текущей ветке.
  Флаги:
    -n, --no-commit — применить изменения, но не создавать коммит
    -x            — добавить в сообщение коммита строку
                    "cherry picked from ..." (полезно для истории)
    --continue, --skip, --abort — работа с конфликтами

5. УДАЛЁННЫЕ РЕПОЗИТОРИИ (REMOTES)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  git remote
  ──────────
  Управление удалёнными репозиториями.
  Флаги:
    -v          — показать адреса (fetch/push) для всех удалённых
                  репозиториев
    add <имя> <url> — добавить новый удалённый репозиторий
    rename <старое> <новое> — переименовать
    remove <имя> — удалить удалённый репозиторий
    set-url <имя> <url> — изменить URL удалённого репозитория
    get-url <имя> — показать URL удалённого репозитория

  git fetch
  ─────────
  Скачивает все изменения из удалённого репозитория, но НЕ
  применяет их к рабочей директории. Просто обновляет
  remote-ветки (например, origin/main).
  Флаги:
    --all       — забрать данные со всех удалённых репозиториев
    --prune     — удалить локальные ссылки на ветки, которых
                  больше нет на удалённом сервере
    --tags      — скачать все теги (по умолчанию теги не тянутся)
    --depth <n> — скачать только указанное количество коммитов

  git pull
  ────────
  Скачивает изменения с удалённого репозитория И применяет их
  к текущей ветке (git fetch + git merge).
  Флаги:
    --rebase    — использовать rebase вместо merge (git fetch +
                  git rebase). Рекомендуется для поддержания
                  чистой истории
    --ff-only   — только fast-forward (откажется, если есть
                  расхождения)
    --squash    — применить все изменения как один коммит
    -r          — сокращение для --rebase
    --all       — выполнить pull для всех веток с remotes

  git push
  ────────
  Отправляет локальные коммиты в удалённый репозиторий.
  Флаги:
    -u, --set-upstream — установить связь между локальной и
                         удалённой веткой (для первого пуша)
    --force     — принудительно отправить (перезаписать историю
                  на сервере). ОПАСНО!
    --force-with-lease — «безопасный форс»: отправит, только если
                         удалённая ветка не была изменена другими
    --tags      — отправить все теги
    --delete    — удалить ветку на сервере
    -n, --dry-run — показать, что будет отправлено, но не отправлять
    --all       — отправить все ветки
    --prune     — удалить на сервере ветки, которых нет локально
    --atomic    — отправить всё или ничего (отмена при любой ошибке)

6. ОТМЕНА ИЗМЕНЕНИЙ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  git restore
  ───────────
  Современная команда для отмены изменений (Git 2.23+).
  Заменила git checkout для файлов.
  Флаги:
    -- <файл>           — отменить изменения в рабочей директории
                          (вернуть файл к состоянию индекса)
    --staged -- <файл>  — убрать файл из индекса (unstage),
                          но оставить изменения в рабочей директории
    --source=<коммит> -- <файл> — восстановить файл до состояния
                                  в указанном коммите
    -W, --worktree      — работать только с рабочей директорией
    -S, --staged        — работать только с индексом

  git reset
  ─────────
  Перемещает HEAD и/или индекс на указанный коммит.
  ОСНОВНЫЕ РЕЖИМЫ:
    --soft     — только переместить HEAD, индекс и файлы
                 не трогать (все изменения остаются в индексе)
    --mixed    — (по умолчанию) переместить HEAD и сбросить индекс,
                 но файлы не трогать (изменения — в рабочей папке)
    --hard     — ПОЛНОСТЬЮ откатить всё: HEAD, индекс и файлы
                 до указанного коммита. НЕОБРАТИМО!
                (сохранённые незакоммиченные изменения пропадут)
    --merge    — сбросить индекс и обновить файлы, но сохранить
                 незакоммиченные изменения
    --keep     — сбросить индекс и обновить файлы, отменяя
                 изменения, только если они не конфликтуют

  git revert <коммит>
  ────────────────────
  Создаёт НОВЫЙ коммит, который отменяет изменения указанного
  коммита. В отличие от reset — не переписывает историю,
  безопасен для общих веток.
  Флаги:
    -n, --no-commit — применить отмену, но не создавать коммит
    --no-edit  — не открывать редактор для сообщения
    --continue, --skip, --abort — работа с конфликтами

  git stash
  ─────────
  Временно прячет незакоммиченные изменения, чтобы можно было
  переключиться на другую ветку. Потом можно вернуть.
  Флаги:
    push       — спрятать изменения (по умолчанию)
    pop        — восстановить последние спрятанные изменения
                  и удалить их из stash
    apply      — восстановить, но НЕ удалять из stash
    list       — показать список всех stash-записей
    show       — показать содержимое последнего stash
    drop       — удалить последний stash
    clear      — удалить все stash-записи
    -u, --include-untracked — спрятать и неотслеживаемые файлы
    -a, --all  — спрятать вообще всё (включая игнорируемые файлы)
    branch <имя> — создать новую ветку из stash и применить его

  git clean
  ─────────
  Удаляет неотслеживаемые файлы из рабочей директории.
  Флаги:
    -n, --dry-run — показать, что будет удалено, но не удалять
    -f, --force   — выполнить удаление (обязательный флаг)
    -d            — удалять и неотслеживаемые папки
    -x            — удалять и игнорируемые (.gitignore) файлы тоже
    -i, --interactive — интерактивный режим

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
7. ТЕГИ (TAGS)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  git tag
  ───────
  Управление тегами (метками) — обычно для отметки релизов
  (v1.0, v2.4.1 и т.д.).
  Флаги:
    <имя>           — создать лёгкий (lightweight) тег на текущем
                      коммите
    -a <имя>        — создать аннотированный тег (с сообщением
                      и автором)
    -m "<сообщение>" — сообщение для аннотированного тега
    -d <имя>        — удалить тег локально
    -l              — список тегов с поиском по шаблону
    --sort=-version:refname — отсортировать по версиям

  git push --tags
  ───────────────
  Отправляет все теги на удалённый репозиторий.

  git push origin --delete <имя_тега>
  ─────────────────────────────────────
  Удаляет тег на удалённом репозитории.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
8. ПРОДВИНУТОЕ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  git bisect
  ──────────
  Бинарный поиск по истории, чтобы найти коммит, в котором
  появился баг.
  Флаги:
    start       — начать поиск
    bad         — пометить текущий коммит как «плохой»
    good <коммит> — пометить коммит как «хороший» (без бага)
    reset       — сбросить bisect после нахождения
    run <скрипт> — автоматический поиск с помощью скрипта
                   (скрипт возвращает 0 = хорошо, 1 = плохо)

  git submodule
  ─────────────
  Управление подмодулями — вложенными репозиториями внутри
  проекта.
  Флаги:
    add <url>   — добавить подмодуль
    update      — обновить все подмодули
    init        — инициализировать подмодули (после клонирования)
    --recursive — рекурсивно применить команду ко всем подмодулям
    --remote    — обновить подмодули до последнего коммита
                  в удалённом репозитории

  git worktree
  ────────────
  Позволяет одновременно работать с несколькими ветками,
  каждую — в своей папке.
  Флаги:
    add <путь> <ветка> — создать новый worktree
    list       — показать все worktree
    remove <путь> — удалить worktree
    lock / unlock — заблокировать worktree от удаления

  git reflog
  ──────────
  Показывает историю ВСЕХ действий в репозитории — включая
  то, что было сброшено reset или потеряно при rebase.
  Это «аварийный журнал» Git.
  Флаги:
    --all       — показать reflog для всех ссылок
    --date=iso  — показать даты в ISO-формате
    expire      — очистить старые записи

  git archive
  ───────────
  Создаёт архив (zip/tar) с файлами из указанного коммита.
  Флаги:
    --format=zip/tar — формат архива
    --output=<файл>  — имя выходного файла
    --prefix=<префикс>/ — добавить префикс к путям внутри архива

  git gc
  ──────
  Запускает сборку мусора — оптимизирует репозиторий
  (сжимает историю, удаляет ненужные объекты).
  Флаги:
    --aggressive — более агрессивная оптимизация (дольше)
    --prune=<дата> — удалять объекты старше указанной даты
    --auto      — запустить только если нужно (вызывается
                  автоматически при некоторых командах)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
9. ПОЛЕЗНЫЕ АЛИАСЫ (СОКРАЩЕНИЯ)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Можно настроить себе сокращения для частых команд:

    git config --global alias.co checkout
    git config --global alias.br branch
    git config --global alias.ci commit
    git config --global alias.st status
    git config --global alias.lg "log --oneline --graph --all"

  Примеры после настройки:
    git lg    →  git log --oneline --graph --all
    git ci -m "фикс"  →  git commit -m "фикс"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
10. ТИПОВОЙ РАБОЧИЙ ПРОЦЕСС (GIT FLOW)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  # Начать новый проект
  git init
  git add .
  git commit -m "Первоначальный коммит"

  # Подключить удалённый репозиторий
  git remote add origin https://github.com/user/repo.git
  git push -u origin main

  # Создать ветку для новой фичи
  git switch -c feature/my-feature

  # Работаем, коммитим
  git add .
  git commit -m "Добавлена новая фича"

  # Забрать изменения из main (пока работали)
  git switch main
  git pull
  git switch feature/my-feature
  git rebase main   # или git merge main

  # Отправить изменения и создать Pull Request
  git push -u origin feature/my-feature

  # После слияния PR — удалить ветку
  git branch -d feature/my-feature
  git push origin --delete feature/my-feature

11. ПОДСКАЗКИ ПО ФЛАГАМ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Комбинирование флагов:
    Большинство коротких флагов можно объединять:
      git log --oneline --graph --all = git log --oneline --graph -all
      git add --all --dry-run = git add -A -n

  -q / --quiet   — тихий режим (минимум вывода)
  -v / --verbose — подробный вывод
  -f / --force   — принудительное выполнение (будьте осторожны!)
  -n / --dry-run — репетиция: показать, что произойдёт,
                   но ничего не делать
  -r             — работа с удалёнными (remote) ссылками
  -a / --all     — применить ко всему / показать всё
  -p             — патч / подробно
